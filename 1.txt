    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <style>
        @import url('https://fonts.googleapis.com/css2?family=Pretendard:wght@400;700;900&display=swap');

    <script>
        tailwind.config = {
          safelist: [
            'text-blue-600', 'border-blue-500', 'bg-blue-50', 'ring-blue-100', 'border-blue-200', 'text-blue-700', 'bg-blue-500', 'text-blue-200',
            'text-green-600', 'border-green-500', 'bg-green-50', 'ring-green-100', 'border-green-200', 'text-green-700', 'bg-green-500', 'text-green-200',
            'text-purple-600', 'border-purple-500', 'bg-purple-50', 'ring-purple-100', 'border-purple-200', 'text-purple-700', 'bg-purple-500', 'text-purple-200',

    <script type="text/babel">
        const { useState } = React;

        const Icons = {
            Baby: (props) => (
                <svg xmlns="http://www.w3.org/2000/svg" width={props.size || 24} height={props.size || 24} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" {...props}>
                    <path d="M9 12h.01"/><path d="M15 12h.01"/><path d="M10 16c.5.3 1.2.5 2 .5s1.5-.2 2-.5"/><path d="M19 6.3a9 9 0 0 1 1.8 3.9 2 2 0 0 1 0 3.6 9 9 0 0 1-17.6 0 2 2 0 0 1 0-3.6A9 9 0 0 1 5 6.3"/><path d="M9 6.7a2 2 0 1 0-4 0a2 2 0 0 0 4 0Z"/><path d="M19 6.7a2 2 0 1 0-4 0a2 2 0 0 0 4 0Z"/>
                </svg>
            ),
            AlertCircle: (props) => (
                <svg xmlns="http://www.w3.org/2000/svg" width={props.size || 24} height={props.size || 24} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" {...props}>
                    <circle cx="12" cy="12" r="10"/><line x1="12" y1="8" x2="12" y2="12"/><line x1="12" y1="16" x2="12.01" y2="16"/>
                </svg>
            ),
            Calendar: (props) => (
                <svg xmlns="http://www.w3.org/2000/svg" width={props.size || 24} height={props.size || 24} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" {...props}>
                    <rect x="3" y="4" width="18" height="18" rx="2" ry="2"/><line x1="16" y1="2" x2="16" y2="6"/><line x1="8" y1="2" x2="8" y2="6"/><line x1="3" y1="10" x2="21" y2="10"/>
                </svg>
            )
        };

        // 달력 표시용 약어 변환기
        const shortenText = (text) => {
            if (!text || text === '-') return '';
            return text
                .replace(/오트밀/g, '오')
                .replace(/현미/g, '현')
                .replace(/보리/g, '보')
                .replace(/퀴노아/g, '퀴')
                .replace(/청경채/g, '청')
                .replace(/애호박/g, '애')
                .replace(/양배추/g, '양')
                .replace(/브로콜리/g, '브')
                .replace(/시금치/g, '시')
                .replace(/당근/g, '당')
                .replace(/단호박/g, '단')
                .replace(/소고기/g, '소')
                .replace(/ /g, '');
        };

        const BabyFoodApp = () => {
            const [activeTab, setActiveTab] = useState('early');
            const [selectedDay, setSelectedDay] = useState(null);

            const START_DATE = new Date(2026, 2, 7); 

            const getFormattedDate = (dayIndex) => {
                const targetDate = new Date(START_DATE);
                targetDate.setDate(START_DATE.getDate() + dayIndex);
                return {
                    dateObj: targetDate,
                    month: targetDate.getMonth() + 1,
                    date: targetDate.getDate(),
                    dayOfWeek: targetDate.getDay(),
                    full: `${targetDate.getMonth() + 1}월 ${targetDate.getDate()}일`,
                    isWeekend: targetDate.getDay() === 0 || targetDate.getDay() === 6,
                    isSunday: targetDate.getDay() === 0,
                    isSaturday: targetDate.getDay() === 6
                };
            };

            const stages = {
                early: {
                    title: "초기 이유식 과정",
                    period: "2026.03.07 ~ 2026.04.05",
                    color: "bg-blue-50 text-blue-700 border-blue-200",
                    accent: "blue",
                    data: Array.from({ length: 30 }, (_, i) => {
                        const day = i + 1;
                        const dateInfo = getFormattedDate(i);
                        let menu = { grain: "쌀 40", meat: "-", toppings: "-", allergy: null };
                        
                        // 이미지 대조 정밀 데이터 반영
                        if (day === 3) menu = { grain: "쌀 20, 오트밀 20", meat: "-", toppings: "-", allergy: null };
                        if (day >= 4 && day <= 6) menu = { grain: "쌀 20, 오트밀 20", meat: "소고기 10", toppings: "-", allergy: null };
                        if (day >= 7 && day <= 9) menu = { grain: "쌀 20, 오트밀 20", meat: "소고기 10", toppings: "청경채 10", allergy: null };
                        if (day >= 10 && day <= 12) menu = { grain: "쌀 20, 오트밀 20", meat: "소고기 10", toppings: "애호박 10", allergy: null };
                        if (day >= 13 && day <= 14) menu = { grain: "쌀 20, 오트밀 20", meat: "소고기 10", toppings: "양배추 10", allergy: null };
                        if (day === 15) menu = { grain: "쌀 20, 오트밀 10, 현미 10", meat: "소고기 10", toppings: "양배추 10", allergy: null };
                        if (day === 16 || day === 17) menu = { grain: "쌀 20, 오트밀 10, 현미 10", meat: "소고기 10", toppings: "양배추 10, 브로콜리 10", allergy: day === 17 ? "밀가루" : null };
                        if (day === 18) menu = { grain: "쌀 20, 오트밀 10, 현미 10", meat: "소고기 10", toppings: "양배추 10, 브로콜리 10", allergy: "밀가루" };
                        if (day >= 19 && day <= 21) menu = { grain: "쌀 20, 오트밀 10, 현미 10", meat: "소고기 10", toppings: "브로콜리 10, 시금치 10", allergy: day === 20 ? "밀가루" : null };
                        if (day >= 22 && day <= 24) menu = { grain: "쌀 20, 오트밀 20, 보리 10", meat: "소고기 10", toppings: "청경채 10, 시금치 10, 당근 10", allergy: (day === 23 || day === 24) ? "노른자" : null };
                        if (day === 25) menu = { grain: "쌀 20, 오트밀 20, 보리 10", meat: "소고기 10", toppings: "청경채 10, 당근 10, 오이 10", allergy: null };
                        if (day === 26 || day === 27) menu = { grain: "쌀 20, 오트밀 20, 보리 10", meat: "소고기 10", toppings: "애호박 10, 당근 10, 오이 10", allergy: "계란" };
                        if (day === 28) menu = { grain: "쌀 20, 오트밀 20, 보리 10", meat: "소고기 10", toppings: "애호박 10, 오이 10, 단호박 10", allergy: null };
                        if (day === 29) menu = { grain: "현미/쌀 20, 오트밀 20, 퀴노아 10", meat: "소고기 10", toppings: "애호박 10, 오이 10, 단호박 10", allergy: null };
                        if (day === 30) menu = { grain: "현미/쌀 20, 오트밀 20, 퀴노아 10", meat: "소고기 10", toppings: "양배추 10, 오이 10, 단호박 10", allergy: "땅콩" };
                        
                        return { day, dateInfo, ...menu };
                    })
                },
                mid1: {
                    title: "중기 1 이유식 과정",
                    period: "초기 종료 후 바로 시작",
                    color: "bg-green-50 text-green-700 border-green-200",
                    accent: "green",
                    data: Array.from({ length: 30 }, (_, i) => ({ day: i + 1, dateInfo: getFormattedDate(i + 30), grain: "잡곡밥 50, 병아리콩 10", meat: "소고기 15, 닭고기 10", toppings: "시금치, 콜리플라워, 새송이", allergy: null }))
                },
                mid2: {
                    title: "중기 2 이유식 과정",
                    period: "중기 1 종료 후 시작",
                    color: "bg-purple-50 text-purple-700 border-purple-200",
                    accent: "purple",
                    data: Array.from({ length: 30 }, (_, i) => ({ day: i + 1, dateInfo: getFormattedDate(i + 60), grain: "잡곡밥 70, 흑미 10", meat: "소고기 15, 대구 10", toppings: "가자미, 파프리카, 연근", allergy: null }))
                }
            };

            const currentStage = stages[activeTab];
            const firstDayOfWeek = currentStage.data[0].dateInfo.dayOfWeek;
            const paddingCells = Array.from({ length: firstDayOfWeek }, (_, i) => i);

            return (
                <div className="min-h-screen p-2 md:p-6 pb-20">
                    <div className="max-w-6xl mx-auto space-y-4">
                        {/* 상단 바 */}
                        <div className="bg-white rounded-3xl p-6 shadow-sm flex flex-col md:flex-row justify-between items-center gap-4 border border-slate-200">
                            <div className="flex items-center space-x-4">
                                <div className="p-3 bg-indigo-500 rounded-2xl text-white shadow-lg shadow-indigo-100">
                                    <Icons.Baby size={32} />
                                </div>
                                <div>
                                    <h1 className="text-xl md:text-2xl font-black text-slate-800 tracking-tight">우리 아기 이유식 달력</h1>
                                    <p className="text-sm text-slate-400 font-medium tracking-tight">3월 7일(토) 시작 스케줄</p>
                                </div>
                            </div>
                            
                            <div className="flex bg-slate-100 p-1 rounded-2xl w-full md:w-auto overflow-hidden">
                                {Object.keys(stages).map((key) => (
                                    <button 
                                        key={key}
                                        onClick={() => { setActiveTab(key); setSelectedDay(null); }}
                                        className={`flex-1 md:flex-none px-6 py-2 rounded-xl text-xs md:text-sm font-bold transition-all ${activeTab === key ? `bg-white text-slate-800 shadow-sm` : 'text-slate-400 hover:text-slate-600'}`}
                                    >
                                        {key === 'early' ? '초기' : key === 'mid1' ? '중기 1' : '중기 2'}
                                    </button>
                                ))}
                            </div>
                        </div>

                        {/* 달력 본체 */}
                        <div className="grid grid-cols-1 lg:grid-cols-12 gap-6">
                            <div className="lg:col-span-12 space-y-4">
                                <div className="bg-white rounded-[2rem] p-3 md:p-6 shadow-sm border border-slate-200">
                                    <div className="flex items-center justify-between mb-6 px-2">
                                        <h2 className={`text-lg font-black text-${currentStage.accent}-600 tracking-tight`}>
                                            {currentStage.title} <span className="text-slate-300 font-normal ml-2 text-sm">{currentStage.period}</span>
                                        </h2>
                                    </div>

                                    {/* 요일 */}
                                    <div className="grid grid-cols-7 mb-2 border-b border-slate-50 pb-2">
                                        {['일', '월', '화', '수', '목', '금', '토'].map((d, i) => (
                                            <div key={d} className={`text-center text-[11px] font-black uppercase tracking-wider ${i === 0 ? 'text-red-400' : i === 6 ? 'text-blue-400' : 'text-slate-400'}`}>
                                                {d}
                                            </div>
                                        ))}
                                    </div>

                                    {/* 그리드 */}
                                    <div className="grid grid-cols-7 gap-1 md:gap-2">
                                        {paddingCells.map(i => (
                                            <div key={`pad-${i}`} className="calendar-cell aspect-square md:aspect-auto bg-slate-50/50 rounded-xl md:rounded-2xl border border-transparent"></div>
                                        ))}
                                        
                                        {currentStage.data.map((item) => (
                                            <button
                                                key={item.day}
                                                onClick={() => setSelectedDay(item)}
                                                className={`calendar-cell relative p-1 md:p-2 rounded-xl md:rounded-2xl border transition-all text-left group overflow-hidden flex flex-col ${
                                                    selectedDay?.day === item.day 
                                                    ? `border-${currentStage.accent}-500 bg-${currentStage.accent}-50 ring-2 ring-${currentStage.accent}-100 shadow-inner` 
                                                    : 'border-slate-100 bg-white hover:border-slate-300'
                                                }`}
                                            >
                                                <div className="flex justify-between items-start mb-0.5 md:mb-1">
                                                    <span className={`text-[10px] md:text-xs font-black ${
                                                        item.dateInfo.isSunday ? 'text-red-500' : 
                                                        item.dateInfo.isSaturday ? 'text-blue-500' : 
                                                        'text-slate-400'
                                                    }`}>
                                                        {item.dateInfo.date}
                                                    </span>
                                                    {item.allergy && (
                                                        <span className="w-1.5 h-1.5 rounded-full bg-red-500 animate-pulse"></span>
                                                    )}
                                                </div>
                                                
                                                {/* 칸 내 표시 (약어 사용) */}
                                                <div className="flex-1 flex flex-col justify-start gap-0.5 md:gap-1 overflow-hidden">
                                                    <div className="text-tiny font-bold text-slate-800 truncate">
                                                        🍚{shortenText(item.grain)}
                                                    </div>
                                                    {item.meat !== '-' && (
                                                        <div className="text-tiny font-bold text-red-600 truncate">
                                                            🥩{shortenText(item.meat)}
                                                        </div>
                                                    )}
                                                    {item.toppings !== '-' && (
                                                        <div className="text-tiny font-medium text-emerald-600 leading-tight">
                                                            🥦{shortenText(item.toppings)}
                                                        </div>
                                                    )}
                                                </div>
                                                
                                                <div className="absolute bottom-0.5 right-1 text-[8px] font-black text-slate-200">
                                                    D{item.day}
                                                </div>
                                            </button>
                                        ))}
                                    </div>
                                </div>
                            </div>
                        </div>

                        {/* 상세 패널 */}
                        {selectedDay && (
                            <div className="fixed bottom-4 left-4 right-4 bg-white rounded-3xl p-5 shadow-2xl border border-slate-200 z-50 animate-in slide-in-from-bottom-10 lg:static lg:shadow-sm lg:animate-none">
                                <div className="flex justify-between items-start mb-4">
                                    <div>
                                        <p className="text-[10px] font-black text-indigo-500 uppercase tracking-widest">Detail View</p>
                                        <h3 className="text-xl font-black text-slate-800 tracking-tight">{selectedDay.dateInfo.full} <span className="text-slate-400 font-bold ml-1 text-base">Day {selectedDay.day}</span></h3>
                                    </div>
                                    <button onClick={() => setSelectedDay(null)} className="p-2 bg-slate-100 rounded-full lg:hidden">✕</button>
                                </div>
                                <div className="grid grid-cols-1 md:grid-cols-3 gap-3">
                                    <div className="p-4 rounded-2xl bg-amber-50 border border-amber-100 shadow-sm">
                                        <p className="text-[11px] font-black text-amber-500 uppercase mb-1">🍚 베이스 곡류</p>
                                        <p className="text-base font-bold text-slate-700 leading-tight">{selectedDay.grain}</p>
                                    </div>
                                    <div className="p-4 rounded-2xl bg-red-50 border border-red-100 shadow-sm">
                                        <p className="text-[11px] font-black text-red-500 uppercase mb-1">🥩 단백질 (고기)</p>
                                        <p className="text-base font-bold text-slate-700 leading-tight">{selectedDay.meat}</p>
                                    </div>
                                    <div className="p-4 rounded-2xl bg-emerald-50 border border-emerald-100 shadow-sm">
                                        <p className="text-[11px] font-black text-emerald-500 uppercase mb-1">🥦 야채 토핑</p>
                                        <p className="text-base font-bold text-slate-700 leading-tight">{selectedDay.toppings}</p>
                                    </div>
                                </div>
                                {selectedDay.allergy && (
                                    <div className="mt-4 p-4 bg-red-600 text-white rounded-2xl flex items-center shadow-lg shadow-red-100">
                                        <Icons.AlertCircle size={20} className="mr-3 shrink-0" />
                                        <div className="text-sm">
                                            <p className="font-black">알레르기 테스트: {selectedDay.allergy}</p>
                                            <p className="text-xs opacity-90 font-medium">오늘은 새로운 재료의 알레르기 반응을 꼭 체크해 주세요!</p>
                                        </div>
                                    </div>
                                )}
                            </div>
                        )}
                    </div>
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<BabyFoodApp />);
    </script>
